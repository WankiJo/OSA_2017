import java.util.Scanner;


public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n =  sc.nextInt();
		int[] a=new int[n];
		double b=1,t=0;
		for(int i=0;i<n;i++){
			a[i]=sc.nextInt();
		}
		for(int j=0;j<n-1;j++){
			if(a[j]>a[j+1]){
				b=a[j];
			}
			else{
				b=a[j+1];
			}
		}
		for(int j=0;j<n;j++){
			t += (a[j]/b)*100;
		}
		System.out.println(t/n);
	}
}
